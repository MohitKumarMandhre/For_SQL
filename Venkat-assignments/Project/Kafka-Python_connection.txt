consumer.py 
-----------------
import kafka

from kafka import KafkaConsumer

consumer = KafkaConsumer("python-test",bootstrap_servers=['localhost:9092'],value_deserializer=lambda x: x.decode('utf-8'))

for msge in consumer:
    print(msge.value)

producer.py
--------------
import kafka
from kafka import KafkaProducer

from time import sleep


producer = KafkaProducer(bootstrap_servers=['localhost:9092'],value_serializer=lambda x: x.encode('utf-8'))

i = 0

while(i<=100):
    producer.send(topic="python-test",value=str(i))
    i+=10
    sleep(2)
    
    
producer.flush()
